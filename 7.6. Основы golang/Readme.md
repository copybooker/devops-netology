2.  Напишите программу, которая найдет наименьший элемент в любом заданном списке, например:

    x := []int{48,96,86,68,57,82,63,70,37,34,83,27,19,97,9,17,}

```
package main

import (
 "fmt"
)

func main() {
 x := []int{48, 96, 86, 68, 57, 82, 63, 70, 37, 34, 83, 27, 19, 97, 9, 17}
 BubbleSort(x)
 fmt.Println(x[0])
}

func BubbleSort(x []int) {
 for i := 0; i < len(x); i++ {
  for j := i; j < len(x); j++ {
   if x[i] > x[j] {
    swap(x, i, j)
   }
  }
 }
}

func swap(x []int, i, j int) {
 tmp := x[i]
 x[i] = x[j]
 x[j] = tmp
}
```
